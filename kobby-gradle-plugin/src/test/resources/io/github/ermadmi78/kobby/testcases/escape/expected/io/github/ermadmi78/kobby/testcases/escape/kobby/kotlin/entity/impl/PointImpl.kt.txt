internal class PointImpl(
  internal val __innerContext: EscapeContext,
  internal val __innerProjection: PointProjectionImpl,
  internal val __innerDto: PointDto
) : Point {
  public override val x: Int
    get() {
      if (!__innerProjection.__innerX0) {
        error("Property [x] is not available - add [x] projection to switch on it")
      }
      return __innerDto.x!!
    }

  public override val y: Int
    get() {
      if (!__innerProjection.__innerY1) {
        error("Property [y] is not available - add [y] projection to switch on it")
      }
      return __innerDto.y!!
    }

  @Suppress("UNUSED_PARAMETER", "UNUSED_CHANGED_VALUE")
  public override fun toString(): String = buildString {
    append("Point").append('(')

    var counter = 0

    if (__innerProjection.__innerX0) {
      if (counter++ > 0) {
        append(", ")
      }
      append("x").append('=').append(x)
    }

    if (__innerProjection.__innerY1) {
      if (counter++ > 0) {
        append(", ")
      }
      append("y").append('=').append(y)
    }

    append(')')
  }

  public override suspend fun query(__projection: QueryProjection.() -> Unit): Query =
      __innerContext.query(__projection)

  public override suspend fun mutation(__projection: MutationProjection.() -> Unit): Mutation =
      __innerContext.mutation(__projection)

  public override fun subscription(__projection: SubscriptionProjection.() -> Unit):
      EscapeSubscriber<Subscription> = __innerContext.subscription(__projection)

  public override fun PointProjection.__withCurrentProjection(): Unit {
    __innerProjection.___innerRepeatProjection(setOf(), this)
  }
}

internal open class PointProjectionImpl : PointProjection {
  internal var __innerX0: Boolean = false

  internal var __innerY1: Boolean = false

  public override fun x(): Unit {
    __innerX0 = true
  }

  public override fun y(): Unit {
    __innerY1 = true
  }

  @Suppress("UNUSED_PARAMETER", "UNUSED_CHANGED_VALUE")
  internal fun ___innerRepeatProjection(ignore: Set<String>, __projection: PointProjection): Unit {
    if ("x" !in ignore && __innerX0) {
      __projection.x()
    }
    if ("y" !in ignore && __innerY1) {
      __projection.y()
    }
  }

  @Suppress("UNUSED_PARAMETER", "UNUSED_CHANGED_VALUE")
  internal fun ___innerBuild(
    ignore: Set<String>,
    header: StringBuilder,
    body: StringBuilder,
    arguments: MutableMap<String, Any>
  ): Unit {
    body.append(' ').append('{')

    // Field: x
    if ("x" !in ignore && __innerX0) {
      body.append(' ').append("x")
    }

    // Field: y
    if ("y" !in ignore && __innerY1) {
      body.append(' ').append("y")
    }

    body.append(' ').append('}')
  }
}

internal fun PointDto.buildEntity(__innerContext: EscapeContext,
    __innerProjection: PointProjectionImpl): Point = PointImpl(__innerContext, __innerProjection,
    this)
